Setup: Types & Traits


    #[derive(Debug, Error)]
    pub enum ServiceError {
        Io(io::Error),
        // ... and a few more.
    }

    pub type ServiceFuture<T>
        = Box<Future<Item=T, Error=ServiceError>>;

    pub trait SmsSender {
        fn send_sms(&self, number: &str, message: &str)
            -> ServiceFuture<()>;
    }

Setup: Models


    #[derive(Clone, Serialize, Deserialize)]
    pub struct User {
        #[serde(rename = "firstName")] pub first_name: String,
        #[serde(rename = "lastName")] pub last_name: String,
        pub number: String,
        #[serde(rename = "createdOn")]
        pub created_on: Option<DateTime<Utc>>,
    }
            

    #[derive(Deserialize)]
    pub struct OneTimePassword {
        pub code: u32,
        pub number: String,
    }

Setup: Calls


    pub struct UsersService {
        repository: UsersRepository,
        sms_sender: Option<Box<SmsSender>>,
    }
            

    impl UsersService {
        /// Validates and registers the user in the repository.
        pub fn register_user(&self, mut data: User)
            -> ServiceFuture<User> { ... }

        /// Authenticate the user using the OTP.
        pub fn verify_otp(&self, otp: OneTimePassword)
            -> ServiceFuture<User> { ... }
    }

Setup: Controller


    let service = Arc::new(UsersService::new());
    App::with_state(service)
        .middleware(middleware::Logger::default())
        .resource("/register", |r| r.method(http::Method::POST)
                                    .with_async(register_user))
        .resource("/verify", |r| r.method(http::Method::POST)
                                  .with_async(verify_otp))
