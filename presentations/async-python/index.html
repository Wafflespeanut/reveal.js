Same code, blocking elsewhere...


  import time
  from threading import Thread

  def work(n):
      time.sleep(n)  # some long running work
      print("Done", n)

  h = [Thread(target=work, args=(n,)) for n in (20, 7, 10, 8)]
  _ = [t.start() for t in h]
  _ = [t.join() for t in h]

Problem?

  • Stack sizes
  • OS and runtime limits
  • On-the-fly creation expensive
  • Expensive context switching

Same code, blocking off-world...


  import os, time
  from multiprocessing import Process

  def work(n):
      time.sleep(n)  # some long running work
      print("Done", n, os.getpid())

  h = [Process(target=work, args=(n,)) for n in (20, 7, 10, 8)]
  _ = [p.start() for p in h]
  _ = [p.join() for p in h]

Problem?

  • Stack sizes
  • IPC overhead
  • On-the-fly creation expensive
  • Expensive context switching

Same code, non-blocking...


  import asyncio

  async def work(n):
      await asyncio.sleep(n)  # some long running work
      print("Done", n)
          

  loop = asyncio.get_event_loop()
  t = [loop.create_task(work(n)) for n in (20, 7, 10, 8)]
  group = asyncio.gather(*t)
  loop.run_until_complete(group)  # block until all finish

How does it work?

  • One giant loop!
  • Tasks submitted to the loop
  • A "piece" of work is done at each step

Perks

Large number of I/O tasks in same thread


  import asyncio, aiohttp

  async def work():
      async with aiohttp.ClientSession() as c:
          async with c.get("https://www.google.com") as resp:
              return len(await resp.text())

  loop = asyncio.get_event_loop()
  t = [loop.create_task(work()) for _ in range(500)]
  ret = loop.run_until_complete(asyncio.gather(*t))
  print(sum(ret))

Possibility to cancel tasks


  import asyncio, aiohttp

  async def work(n):
      await asyncio.sleep(n)
      print("Done", n)

  loop = asyncio.get_event_loop()
  t1 = loop.create_task(work(10))
  t2 = loop.create_task(work(5))
  loop.run_until_complete(t2)
  t1.cancel()

Common pitfalls

Forgetting to "await"


  import asyncio

  async def work(n):
      asyncio.sleep(n)  # Oops...
      print("Done", n)

  loop = asyncio.get_event_loop()
  t = [loop.create_task(work(n)) for n in (20, 7, 10, 8)]
  loop.run_until_complete(asyncio.gather(*t))

Sync call inside async function...


  import asyncio, time

  async def work(n):
      time.sleep(n)  # Uh oh...
      print("Done", n)

  loop = asyncio.get_event_loop()
  t = [loop.create_task(work(n)) for n in (20, 7, 10, 8)]
  loop.run_until_complete(asyncio.gather(*t))

Solution: Use threads!


  import asyncio, time
  from concurrent.futures.thread import ThreadPoolExecutor

  loop = asyncio.get_event_loop()
  pool = ThreadPoolExecutor(max_workers=8)
          

  async def work(n):
      await loop.run_in_executor(pool, time.sleep, n)
      print("Done", n)

  t = [loop.create_task(work(n)) for n in (20, 7, 10, 8)]
  loop.run_until_complete(asyncio.gather(*t))

  import asyncio, time
  from concurrent.futures.thread import ProcessPoolExecutor

  loop = asyncio.get_event_loop()
  pool = ProcessPoolExecutor(max_workers=8)
          

  def my_sum(n):
      return sum(i ** 10 for i in range(int(n)))

  async def work(n):
      result = await loop.run_in_executor(pool, my_sum, n)
      print("Done", result)

  t = [loop.create_task(work(n)) for n in (5e6, 3e6, 8e6, 2e6)]
  loop.run_until_complete(asyncio.gather(*t))
