Why Rust?

C/C++ problems: dangling pointers


    int *some_dumb_function() {
        int a = 2;
        return &a;      // return reference to a local variable
    }

    void foo() {
        int *raw_ptr = some_dumb_function();
        *raw_ptr = 3;   // UB or segfault?
    }

C/C++ problems: Use after free


    void *another_dumb_function() {
        int *ptr = malloc(sizeof(int));
        free(ptr);
        printf("%d", *ptr);   // uh oh! freed memory!
    }

C/C++ problems: Double free


    void *another_dumb_function() {
        int *ptr = malloc(sizeof(int));
        free(ptr);
        free(ptr);    // wheee...
    }
